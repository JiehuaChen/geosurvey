# clone a repository:
git clone [git-url]

# review changes
git status
git diff

# commit your changes:
git commit -a -m "short message describing what i changed"

# push your changes: 
git push origin master

# pull the latest changes:
git pull origin master

# create a new branch 
git checkout -b [name-of-branch]

# find which files are under version control
git ls-files

# who to blame for every line in a program
git blame [filename]

# adjust attribution
git config user.name [name]
git config user.email [email]
